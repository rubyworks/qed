def clean_text
  str = text.chomp.sub(/\A\n/,'')
  if md = /\A["]{3,}(.*?)["]{3,}\Z/.match(str)
    str = md[1]
  end
  str.rstrip
end

# File lib/qed/parser.rb, line 209
def ready!(flush, back_step)
  @flush     = flush
  @back_step = back_step

  @text  = raw.map{ |lineno, line| line }.join
  @type  = parse_type

  @back_step.next_step = self if @back_step

  self
end
